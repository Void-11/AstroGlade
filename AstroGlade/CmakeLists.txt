add_executable(${ASTRO_GLADE_GAME_TARGET_NAME}
    ${CMAKE_CURRENT_SOURCE_DIR}/src/main.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/include/framework/Application.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/framework/Application.cpp
)

target_include_directories(${ASTRO_GLADE_GAME_TARGET_NAME} PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(${ASTRO_GLADE_GAME_TARGET_NAME} PUBLIC sfml-graphics)
target_link_libraries(${ASTRO_GLADE_GAME_TARGET_NAME} PUBLIC sfml-window)
target_link_libraries(${ASTRO_GLADE_GAME_TARGET_NAME} PUBLIC sfml-system)
target_link_libraries(${ASTRO_GLADE_GAME_TARGET_NAME} PUBLIC sfml-audio)

function(CopyLibToTarget LIB_NAME TARGET_NAME)
add_custom_command(
    TARGET ${TARGET_NAME}
    POST_BUILD
    COMMAND
    ${CMAKE_COMMAND} -E copy_if_different
    $<TARGET_FILE:${LIB_NAME}>
    $<TARGET_FILE_DIR:${ASTRO_GLADE_GAME_TARGET_NAME}>
)
endfunction()

CopyLibToTarget(sfml-graphics ${ASTRO_GLADE_GAME_TARGET_NAME})
CopyLibToTarget(sfml-window ${ASTRO_GLADE_GAME_TARGET_NAME})
CopyLibToTarget(sfml-system ${ASTRO_GLADE_GAME_TARGET_NAME})
CopyLibToTarget(sfml-audio ${ASTRO_GLADE_GAME_TARGET_NAME})
