cmake_minimum_required(VERSION 3.27.7)

project(AstroGlade
	VERSION 1.0.0
	LANGUAGES C CXX
)

set(ASTRO_GLADE_ENGINE_TARGET_NAME AstroGladeEngine) 
set(ASTRO_GLADE_GAME_TARGET_NAME AstroGlade) 

add_subdirectory(AstroGladeEngine)
add_subdirectory(AstroGlade)
